Quick Start

Get up and running with a simple classification pipeline.

library(VectorForgeML)

# Load data
data(iris)

# Create pipeline
pipe <- Pipeline(steps = list(
    c("scaler", StandardScaler()),
    c("model", SoftmaxRegression())
))

# Train
pipe$fit(iris[, 1:4], iris[, 5])
