# Create a pipeline
pipe <- Pipeline(
  steps = list(
    c("scaler", StandardScaler()),
    c("pca", PCA(n_components = 2)),
    c("model", LogisticRegression())
  )
)

# Train in milliseconds
pipe$fit(X_train, y_train)
preds <- pipe$predict(X_test)
